
@Configuration annotation is part of the spring core framework.
 Spring Configuration annotation indicates that this class has @Beans definition
 methods. So Spring Container can process the class and generate Spring Beans
 to be used by the application

ProjectConfig.java

```java
package com.spring.demo.config;

import com.spring.demo.beans.Vehicle;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

@Configuration
public class ProjectConfig {

    @Bean
    Vehicle vehicle(){
        Vehicle vehicle = new Vehicle();
        vehicle.setName("David putar");
        return vehicle;
    }

    @Bean
    String hello(){
        return "Hello world";
    }
}

```

Vehicle.java

```java
package com.spring.demo.beans;

public class Vehicle {
    public String getName() {
        return name;
    }

    public void setName(String name) {
        this.name = name;
    }

    private String name;

}
```

Main.java

```java
import com.spring.demo.beans.Vehicle;
import com.spring.demo.config.ProjectConfig;
import org.springframework.context.annotation.AnnotationConfigApplicationContext;

void main(){

    var context = new AnnotationConfigApplicationContext(ProjectConfig.class);

    /*
    we don't need to do any explicit casting while fetching a bean from context.
    spring is smart enough to look for a bean of the type you requested in its context
    If such bean doesn't exist, spring will throw an exception
     */

    Vehicle veh = context.getBean(Vehicle.class);
    System.out.println(veh.getName());

    String hello = context.getBean(String.class);
    System.out.println(hello);

    String hello1 = (String) context.getBean("hello");
    System.out.println(hello1);

}

```
