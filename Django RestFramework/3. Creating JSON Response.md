views.py
```python
from django.shortcuts import render
from .models import Movie
from django.http import JsonResponse
# Create your views here.
def movie_list(request):
    movies = Movie.objects.all()
    print(movies) #complex queryset
    print(movies.values()) #queryset with list of dictionary
    print(list(movies.values())) #list of dictionary
    return JsonResponse({'movies':list(movies.values())}) #json response with list of dictionary

def movie_details(request, pk):
    movie = Movie.objects.get(pk=pk)
    return JsonResponse({
        'name': movie.name,
        'description': movie.description,
        'active': movie.active
    })
```
